cpp-HashTable

2024-06-01
#include <vector>
#include <list>

template <typename K, typename V>
class HashTable {
public:
    HashTable(int size = 100) : table_(size) {}

    void insert(const K& key, const V& value) {
        int index = hash(key) % table_.size();
        table_[index].push_back(std::make_pair(key, value));
    }

    V get(const K& key) {
        int index = hash(key) % table_.size();
        for (auto& pair : table_[index]) {
            if (pair.first == key) {
                return pair.second;
            }
        }
        throw std::out_of_range("Key not found");
    }

    bool contains(const K& key) {
        int index = hash(key) % table_.size();
        for (auto& pair : table_[index]) {
            if (pair.first == key) {
                return true;
            }
        }
        return false;
    }

private:
    std::vector<std::list<std::pair<K, V>>> table_;

    int hash(const K& key) {
        // 这里可以根据实际情况实现哈希函数
        return std::hash<K>()(key);
    }
};
