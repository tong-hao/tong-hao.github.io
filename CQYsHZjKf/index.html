func max(num1, num2 int) int{
   /* 声明局部变量 */
   var result int

   if (num1 > num2) {
      result = num1
   } else {
      result = num2
   }
   return result 
}

反射

TypeOf & ValueOf

import "reflect"

var isbook bool = true
reflect.TypeOf(isbook)
reflect.ValueOf(isbook)


func setvalue1(x interface{}){
	v := reflect.ValueOf(x)
	if value.Kind() == reflect.String {
		value.SetString("changed")
	}
}

// 指针
func setvalue2(x interface{}){
	v := reflect.ValueOf(x)
	if value.Elem().Kind() == reflect.String {
		value.Elem().SetString("changed")
	}
}

str1 := "hello"
setvalue1(str1)
setvalue2(&str1)

sync.WaitGroup

import ("fmt"
				"sync"
)

var wg sync.WaitGroup
func hello(){
	fmt.Println("hello")
	defer wg.Done() // -1
}

wg.Add(1) // +1
go hello()
wg.Wait() // 阻塞，直到为0
